# Azure VM Deployment Commands
# Copy and paste these commands in your Azure VM console

# 1. Clone the repository fresh
git clone https://github.com/Scorpion07/chathere.git
cd chathere

# 2. Set Environment Variables with your API keys
export GOOGLE_API_KEY="YOUR_GOOGLE_API_KEY_HERE"
export OPENAI_API_KEY="YOUR_OPENAI_API_KEY_HERE"
export DEEPSEEK_API_KEY="YOUR_DEEPSEEK_API_KEY_HERE"
export ANTHROPIC_API_KEY="YOUR_ANTHROPIC_API_KEY_HERE"

# 3. Make environment variables persistent
echo 'export GOOGLE_API_KEY="YOUR_GOOGLE_API_KEY_HERE"' >> ~/.bashrc
echo 'export OPENAI_API_KEY="YOUR_OPENAI_API_KEY_HERE"' >> ~/.bashrc
echo 'export DEEPSEEK_API_KEY="YOUR_DEEPSEEK_API_KEY_HERE"' >> ~/.bashrc
echo 'export ANTHROPIC_API_KEY="YOUR_ANTHROPIC_API_KEY_HERE"' >> ~/.bashrc

# 4. Reload environment
source ~/.bashrc

# 5. Install Python dependencies
pip install -r requirements-clean.txt

# 6. Build frontend (if Node.js is installed)
cd frontend
npm install
npm run build
cd ..

# 7. Start the application
gunicorn -c gunicorn.conf.py wsgi:app
